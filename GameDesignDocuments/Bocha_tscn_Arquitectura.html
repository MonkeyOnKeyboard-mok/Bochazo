3.1 Bocha.gd (Root - Node3D)

Responsabilidad: Nodo raíz que actúa como orquestador. Referencia a los componentes hijos y expone señales globales de la bocha. 
# Bocha.gd - Nodo Raíz
extends Node3D

# Señales para comunicación externa
signal bocha_thrown(power: float, direction: Vector3)
signal bocha_stopped
signal state_changed(new_state: String)

# Referencias a componentes hijos (asignadas en _ready)
var physics_body: RigidBody3D
var state_machine: Node
var position_controller: Node
var throw_controller: Node
var input_handler: Node
var visual_effects: Node3D

# Configuración exportable
@export_group("Configuración")
@export var lateral_speed: float = 3.0
@export var lateral_bounds: Vector2 = Vector2(-2.0, 2.0)
@export var max_throw_power: float = 15.0
@export var throw_sensitivity: float = 1.0

func _ready() -> void:
    # Cache de referencias a hijos
    physics_body = $PhysicsBody
    state_machine = $StateMachine
    position_controller = $PositionController
    throw_controller = $ThrowController
    input_handler = $InputHandler
    visual_effects = $VisualEffects

    # Inicializar componentes con referencias cruzadas
    position_controller.bocha_root = self
    throw_controller.bocha_root = self
    input_handler.bocha_root = self

    # Conectar señales internas
    state_machine.state_changed.connect(_on_state_changed)

func _on_state_changed(new_state: String) -> void:
    state_changed.emit(new_state)

3.2 StateMachine.gd (Node)

Responsabilidad: Gestiona las transiciones entre estados usando un patrón State Machine. Cada estado es un Resource con su lógica encapsulada. 
# StateMachine.gd
extends Node

signal state_changed(new_state: String)

# Estados disponibles
enum BochaState {
    POSITIONING,  # Movimiento lateral para ubicarse
    AIMING,       # Preparando el lanzamiento (mouse gesture)
    THROWING,     # Aplicando impulso
    ROLLING       # Bocha rodando libremente
}

var current_state: BochaState = BochaState.POSITIONING
var states: Dictionary = {}

func _ready() -> void:
    # Registrar estados
    states[BochaState.POSITIONING] = PositioningState.new()
    states[BochaState.AIMING] = AimingState.new()
    states[BochaState.THROWING] = ThrowingState.new()
    states[BochaState.ROLLING] = RollingState.new()

    # Inicializar cada estado con referencia al padre
    for state in states.values():
        state.init(get_parent())

    # Entrar al estado inicial
    states[current_state].enter()

func _process(delta: float) -> void:
    states[current_state].update(delta)

func _physics_process(delta: float) -> void:
    states[current_state].physics_update(delta)

func _input(event: InputEvent) -> void:
    states[current_state].handle_input(event)

func transition_to(new_state: BochaState) -> void:
    if new_state == current_state:
        return

    states[current_state].exit()
    current_state = new_state
    states[current_state].enter()
    state_changed.emit(BochaState.keys()[current_state])

    pass

    if event is InputEventMouseButton and event.pressed:
        if event.button_index == MOUSE_BUTTON_LEFT:
            # Verificar si el click fue sobre la bocha (raycast)
            if bocha.input_handler.is_clicking_on_bocha(event.position):

3.5 ThrowingState.gd (Resource)

Responsabilidad: Aplica el impulso calculado al RigidBody3D y transiciona a ROLLING. 
# ThrowingState.gd
extends RefCounted
class_name ThrowingState

var bocha: Node3D
var throw_applied: bool = false

func init(root: Node3D) -> void:
    bocha = root

func enter() -> void:
    throw_applied = false

func exit() -> void:
    pass

func update(_delta: float) -> void:
    if not throw_applied:
        var throw_data = bocha.throw_controller.get_throw_data()

        # Aplicar impulso al RigidBody3D
        var impulse = throw_data.direction * throw_data.power
        bocha.physics_body.apply_central_impulse(impulse)

        # Efecto de curva según movimiento lateral del mouse
        if abs(throw_data.curve) > 0.01:
            bocha.physics_body.angular_velocity.y = throw_data.curve * 2.0

        bocha.bocha_thrown.emit(throw_data.power, throw_data.direction)
        bocha.visual_effects.play_throw_effect()

        throw_applied = true

func physics_update(_delta: float) -> void:
    if throw_applied:
        # Transicionar a ROLLING después de aplicar impulso
        bocha.state_machine.transition_to(
            bocha.state_machine.BochaState.ROLLING
        )

func handle_input(_event: InputEvent) -> void:
    pass

3.6 RollingState.gd (Resource)

Responsabilidad: Monitorea la bocha mientras rueda. Cuando se detiene (velocidad cercana a 0), emite señal y transiciona a POSITIONING. 
# RollingState.gd
extends RefCounted
class_name RollingState

var bocha: Node3D
var stop_threshold: float = 0.05
var time_below_threshold: float = 0.0
var required_stop_time: float = 0.5

func init(root: Node3D) -> void:
    bocha = root

func enter() -> void:
    time_below_threshold = 0.0
    bocha.physics_body.freeze = false

func exit() -> void:
    pass

func update(_delta: float) -> void:
    # Efecto de trail mientras rueda
    var speed = bocha.physics_body.linear_velocity.length()
    if speed > 0.5:
        bocha.visual_effects.update_trail(speed)

func physics_update(delta: float) -> void:
    var speed = bocha.physics_body.linear_velocity.length()

    if speed < stop_threshold:
        time_below_threshold += delta
        if time_below_threshold >= required_stop_time:
            bocha.physics_body.linear_velocity = Vector3.ZERO
            bocha.physics_body.angular_velocity = Vector3.ZERO
            bocha.bocha_stopped.emit()
            bocha.state_machine.transition_to(
                bocha.state_machine.BochaState.POSITIONING
            )
    else:
        time_below_threshold = 0.0

func handle_input(_event: InputEvent) -> void:
    pass

3.7 InputHandler.gd (Node)

Responsabilidad: Maneja la detección de clicks sobre la bocha usando raycasting desde la cámara. 
# InputHandler.gd
extends Node

var bocha_root: Node3D
var ray_query: PhysicsRayQueryParameters3D

func is_clicking_on_bocha(mouse_position: Vector2) -> bool:
    var camera = get_viewport().get_camera_3d()
    if camera == null:
        return false

    # Crear rayo desde la cámara hacia la posición del mouse
    var ray_origin = camera.project_ray_origin(mouse_position)
    var ray_end = ray_origin + camera.project_ray_normal(mouse_position) * 100.0

    ray_query = PhysicsRayQueryParameters3D.create(
        ray_origin, ray_end, 1 << 0  # collision mask 1
    )
    ray_query.collide_with_areas = false
    ray_query.collide_with_bodies = true

    var space_state = bocha_root.physics_body.get_world_3d().direct_space_state
    var result = space_state.intersect_ray(ray_query)

    if result.is_empty():
        return false

    # Verificar si el collider pertenece a esta bocha
    return result.collider == bocha_root.physics_body

3.8 PositionController.gd (Node)

Responsabilidad: Controla el movimiento lateral de la bocha durante el estado POSITIONING. Usa las acciones de input configuradas en el proyecto. 
# PositionController.gd
extends Node

var bocha_root: Node3D

func update_position() -> void:
    var direction = 0.0

    if Input.is_action_pressed("move_left"):
        direction += 1.0
    if Input.is_action_pressed("move_right"):
        direction -= 1.0

    if direction == 0.0:
        return

    var speed = bocha_root.lateral_speed * direction
    var new_x = bocha_root.physics_body.position.x + speed * get_process_delta_time()

    # Respetar límites laterales
    new_x = clamp(new_x, bocha_root.lateral_bounds.x, bocha_root.lateral_bounds.y)

    # Mover solo en X (lateral), mantener Y y Z
    var new_pos = bocha_root.physics_body.position
    new_pos.x = new_x
    bocha_root.physics_body.position = new_pos

3.9 ThrowController.gd (Node) ⭐ NODO CLAVE

Responsabilidad: Detecta el gesto del mouse (hacia atrás y luego hacia adelante) para calcular la potencia, dirección y efecto (curva) del lanzamiento. Este es el corazón de la mecánica de lanzamiento. 
# ThrowController.gd
extends Node

var bocha_root: Node3D

# Estado del gesto
enum GesturePhase {
    IDLE,           # Sin gesto activo
    PULL_BACK,      # Mouse moviéndose hacia atrás
    PUSH_FORWARD,   # Mouse moviéndose hacia adelante
    RELEASED        # Gesto completado
}

var gesture_phase: GesturePhase = GesturePhase.IDLE

# Datos del gesto
var pull_start_pos: Vector2 = Vector2.ZERO
var pull_max_distance: float = 0.0
var push_velocity: Vector2 = Vector2.ZERO
var previous_mouse_pos: Vector2 = Vector2.ZERO
var gesture_start_time: float = 0.0

# Historial de posiciones para calcular velocidad
var mouse_history: Array = []
const MAX_HISTORY: int = 10

# Configuración
@export var min_pull_distance: float = 50.0   # Pixeles mínimos hacia atrás
@export var max_pull_distance: float = 500.0  # Pixeles máximos
@export var sensitivity: float = 1.0

func start_aiming() -> void:
    gesture_phase = GesturePhase.IDLE
    mouse_history.clear()
    pull_max_distance = 0.0
    Input.set_mouse_mode(Input.MOUSE_MODE_VISIBLE)

func stop_aiming() -> void:
    gesture_phase = GesturePhase.IDLE
    mouse_history.clear()

func update_aim() -> void:
    var current_mouse = get_viewport().get_mouse_position()
    var viewport_height = get_viewport().size.y

    # Agregar al historial
    mouse_history.append({
        "position": current_mouse,
        "time": Time.get_ticks_msec()
    })
    if mouse_history.size() > MAX_HISTORY:
        mouse_history.pop_front()

    match gesture_phase:
        GesturePhase.IDLE:
            # Detectar inicio del gesto (mouse moviéndose hacia abajo/atrás)
            if previous_mouse_pos != Vector2.ZERO:
                var delta_y = current_mouse.y - previous_mouse_pos.y
                if delta_y > 5.0:  # Movimiento hacia abajo detectado
                    gesture_phase = GesturePhase.PULL_BACK
                    pull_start_pos = current_mouse
                    gesture_start_time = Time.get_ticks_msec()
                    pull_max_distance = 0.0

        GesturePhase.PULL_BACK:
            # Calcular distancia hacia atrás
            var distance = current_mouse.y - pull_start_pos.y
            if distance > 0:
                pull_max_distance = max(pull_max_distance, distance)

                # Detectar cambio de dirección (empieza a subir)
                if mouse_history.size() >= 3:
                    var recent_avg_y = _get_average_y(3)
                    var older_avg_y = _get_average_y(6, 3)
                    if recent_avg_y < older_avg_y and pull_max_distance >= min_pull_distance:
                        gesture_phase = GesturePhase.PUSH_FORWARD
                        mouse_history.clear()

        GesturePhase.PUSH_FORWARD:
            # Calcular velocidad del push hacia adelante
            if mouse_history.size() >= 3:
                var recent = mouse_history[-1]
                var older = mouse_history[-3]
                var time_diff = recent["time"] - older["time"]

                if time_diff > 0:
                    push_velocity = (recent["position"] - older["position"]) / time_diff * 16.0

            # Detectar si el push se completó (velocidad baja o mouse se detiene)
            if mouse_history.size() >= 5:
                var recent_speed = _get_recent_speed()
                if recent_speed < 50.0 or push_velocity.y < -20.0:
                    gesture_phase = GesturePhase.RELEASED

    previous_mouse_pos = current_mouse

func get_throw_data() -> Dictionary:
    var data = {
        "power": 0.0,
        "direction": Vector3.FORWARD,
        "curve": 0.0
    }

    if gesture_phase != GesturePhase.RELEASED:
        return data

    # Calcular potencia basada en la velocidad del push
    var forward_speed = abs(push_velocity.y)
    var normalized_power = clamp(forward_speed / 200.0, 0.0, 1.0)

    # También considerar la distancia del pull-back como factor
    var pull_factor = clamp(pull_max_distance / max_pull_distance, 0.0, 1.0)

    # Combinar ambos factores
    data.power = (normalized_power * 0.7 + pull_factor * 0.3) * bocha_root.max_throw_power

    # Dirección principal (hacia adelante en el eje Z del mundo)
    data.direction = Vector3.FORWARD

    # Efecto de curva basado en movimiento lateral del mouse durante el push
    var lateral_movement = push_velocity.x
    data.curve = clamp(lateral_movement / 100.0, -1.0, 1.0) * sensitivity

    return data

# Helpers
func _get_average_y(count: int, offset: int = 0) -> float:
    if mouse_history.size() < count + offset:
        return 0.0
    var total = 0.0
    for i in range(offset, offset + count):
        total += mouse_history[i]["position"].y
    return total / count

func _get_recent_speed() -> float:
    if mouse_history.size() < 2:
        return 0.0
    var recent = mouse_history[-1]
    var older = mouse_history[-2]
    var time_diff = recent["time"] - older["time"]
    if time_diff == 0:
        return 0.0
    return recent["position"].distance_to(older["position"]) / time_diff * 16.0

7. Ventajas de esta Arquitectura

  • Composición pura: Cada nodo tiene una responsabilidad única y bien definida.
  • State Machine: Los estados son Resources, fáciles de testear y modificar independientemente.
  • ThrowController desacoplado: La lógica de detección de gestos está aislada, se puede ajustar sin tocar física ni estados.
  • Señales claras: La comunicación entre nodos usa señales, no referencias directas innecesarias.
  • Extensible: Agregar un nuevo estado (ej: "CurveState" para efecto de curva) es crear un nuevo Resource.
  • Testeable: Cada componente puede probarse aisladamente.
  • Reutilizable: PositionController o ThrowController pueden usarse en otros objetos del juego.

8. Mecánica de Lanzamiento - Detalle del Gesto

El gesto de lanzamiento funciona en 3 fases:

Fase 1: PULL_BACK (Hacia atrás)

El usuario mueve el mouse hacia abajo (hacia atrás en pantalla). Se registra la posición inicial y se mide la distancia máxima alcanzada. Esta distancia contribuye a la potencia del lanzamiento.

min_pull_distance = 50px | max_pull_distance = 500px

Fase 2: PUSH_FORWARD (Hacia adelante)

El usuario cambia de dirección y mueve el mouse hacia arriba (hacia adelante). Se mide la velocidad del movimiento. Esta velocidad es el factor principal de potencia.

power = (velocidad_push * 0.7 + distancia_pull * 0.3) * max_throw_power

Fase 3: RELEASED (Suelta)

Cuando la velocidad del mouse baja significativamente (el usuario "suelta" el gesto), se calculan los datos finales del lanzamiento y se aplica el impulso.

curve = movimiento_lateral_mouse / 100.0 → Se aplica como angular_velocity.y

9. Los Resources como Estados - Explicación Detallada

9.1 ¿Qué son y por qué Resources?

En Godot, un Resource es un objeto de datos que puede ser instanciado, serializado y compartido. En este patrón, cada estado de la bocha es una clase que extiende RefCounted (la base de los resources en Godot) y define un class_name para que sea reconocible globalmente.

¿Por qué no usar nodos para los estados? Porque los estados no necesitan estar en el SceneTree. No necesitan _process, _ready, ni transform. Son pura lógica. Usar Resources los hace:
  • Livianos: No tienen overhead de nodo.
  • Testeables: Podés instanciar un estado y probarlo sin la escena completa.
  • Serializable: Podés guardarlos como .tres y configurar propiedades desde el inspector.
  • Reutilizables: El mismo Resource puede usarse en múltiples bochas.

9.2 Interfaz que cumple cada State Resource

Todos los estados comparten la misma interfaz implícita (no hay interfaz formal en GDScript, pero se respeta el contrato): 
# Cada estado DEBE implementar estos 5 métodos:

func init(root: Node3D) -> void:
    # Se llama UNA vez al crear la StateMachine.
    # Recibe la referencia al nodo raíz Bocha para acceder a sus componentes.

func enter() -> void:
    # Se llama CADA vez que se entra a este estado.
    # Acá se configura el entorno: freeze physics, mostrar UI, etc.

func update(delta: float) -> void:
    # Se llama en cada _process() mientras este estado esté activo.
    # Lógica que no necesita physics (UI, input visual, etc).

func physics_update(delta: float) -> void:
    # Se llama en cada _physics_process() mientras este estado esté activo.
    # Lógica de física: chequear velocidad, aplicar fuerzas, etc.

func handle_input(event: InputEvent) -> void:
    # Se llama en cada _input() mientras este estado esté activo.
    # Detecta clicks, teclas, gestos y decide si transicionar.

func exit() -> void:
    # Se llama CADA vez que se sale de este estado.
    # Cleanup: unfreeze physics, ocultar UI, resetear variables.

9.3 Cómo funciona el ciclo de vida

El flujo de un estado es siempre:

init() (una vez al inicio)
→
enter()
→
update() + physics_update() + handle_input() (loop cada frame)
→
exit()
→
enter() (del siguiente estado)

Estado enter() update() / physics_update() handle_input() exit() Transiciona a

9.5 ¿Se guardan como archivos .tres?

No es obligatorio. Hay dos formas de usar estos Resources:

Opción A - Instanciados por código (la que usamos acá):
El StateMachine los crea con .new() en su _ready(). No hay archivos .tres. Es más simple y no requiere configuración manual.

Opción B - Archivos .tres configurables:
Cada estado se guarda como .tres y se asigna desde el inspector del StateMachine. Esto permite ajustar propiedades (como stop_threshold en RollingState) sin tocar código.

Recomendación: Empezar con Opción A (código). Si después necesitás ajustar valores frecuentemente, migrar a Opción B.

11. Opción B: Configurar los States como Resources .tres desde el Inspector

11.1 ¿Qué cambia respecto a la Opción A?

En la Opción A el StateMachine crea los estados con .new() en código. Todo está hardcodeado.
En la Opción B cada estado es un archivo .tres independiente que se asigna desde el inspector del StateMachine. Esto permite:
  • Ajustar valores sin tocar código: cada estado tiene propiedades @export visibles en el inspector.
  • Tener variantes: podés crear "RollingState_rapido.tres" y "RollingState_lento.tres" con distintos thresholds.
  • Previsualizar: al abrir un .tres en el inspector ves todas sus propiedades.

11.2 Paso 1: Modificar los scripts de los States para agregar @export

Cada state resource necesita propiedades exportables. Acá va cómo queda cada uno:

PositioningState.gd

extends RefCounted
class_name PositioningState

@export_group("Configuración")
@export var freeze_on_enter: bool = true
@export var reset_velocity_on_enter: bool = true

var bocha: Node3D

func init(root: Node3D) -> void:
    bocha = root

func enter() -> void:
    if freeze_on_enter:
        bocha.physics_body.freeze = true
    if reset_velocity_on_enter:
        bocha.physics_body.linear_velocity = Vector3.ZERO

func exit() -> void:
    bocha.physics_body.freeze = false

func update(_delta: float) -> void:
    bocha.position_controller.update_position()

func physics_update(_delta: float) -> void:
    pass

func handle_input(event: InputEvent) -> void:
    if event is InputEventMouseButton and event.pressed:
        if event.button_index == MOUSE_BUTTON_LEFT:
            if bocha.input_handler.is_clicking_on_bocha(event.position):
                bocha.state_machine.transition_to(
                    bocha.state_machine.BochaState.AIMING
                )

extends RefCounted
class_name AimingState

@export_group("Configuración")
@export var show_power_indicator: bool = true
@export var min_power_to_throw: float = 0.1

var bocha: Node3D

func init(root: Node3D) -> void:

func exit() -> void:

    bocha.visual_effects.hide_power_indicator()

    pass

func handle_input(event: InputEvent) -> void:
    if event is InputEventMouseButton and event.pressed:
        if event.button_index == MOUSE_BUTTON_LEFT:
            var throw_data = bocha.throw_controller.get_throw_data()
            if throw_data.power >= min_power_to_throw:

ThrowingState.gd

extends RefCounted
class_name ThrowingState

@export_group("Configuración")
@export var apply_curve: bool = true
@export var curve_multiplier: float = 2.0
@export var play_throw_effect: bool = true

var bocha: Node3D
var throw_applied: bool = false

func init(root: Node3D) -> void:
    bocha = root

func enter() -> void:
    throw_applied = false

func exit() -> void:
    pass

func update(_delta: float) -> void:
    if not throw_applied:
        var throw_data = bocha.throw_controller.get_throw_data()
        var impulse = throw_data.direction * throw_data.power
        bocha.physics_body.apply_central_impulse(impulse)

        if apply_curve and abs(throw_data.curve) > 0.01:
            bocha.physics_body.angular_velocity.y = throw_data.curve * curve_multiplier

        bocha.bocha_thrown.emit(throw_data.power, throw_data.direction)
        if play_throw_effect:
            bocha.visual_effects.play_throw_effect()

        throw_applied = true

func physics_update(_delta: float) -> void:
    if throw_applied:
        bocha.state_machine.transition_to(
            bocha.state_machine.BochaState.ROLLING
        )

func handle_input(_event: InputEvent) -> void:
    pass

RollingState.gd

extends RefCounted
class_name RollingState

@export_group("Configuración")
@export var stop_threshold: float = 0.05
@export var required_stop_time: float = 0.5
@export var enable_trail_effect: bool = true
@export var trail_min_speed: float = 0.5

var bocha: Node3D
var time_below_threshold: float = 0.0

func init(root: Node3D) -> void:
    bocha = root

func enter() -> void:
    time_below_threshold = 0.0
    bocha.physics_body.freeze = false

func exit() -> void:
    pass

func update(_delta: float) -> void:
    if not enable_trail_effect:
        return
    var speed = bocha.physics_body.linear_velocity.length()
    if speed > trail_min_speed:
        bocha.visual_effects.update_trail(speed)

func physics_update(delta: float) -> void:
    var speed = bocha.physics_body.linear_velocity.length()
    if speed < stop_threshold:
        time_below_threshold += delta
        if time_below_threshold >= required_stop_time:
            bocha.physics_body.linear_velocity = Vector3.ZERO
            bocha.physics_body.angular_velocity = Vector3.ZERO
            bocha.bocha_stopped.emit()
            bocha.state_machine.transition_to(
                bocha.state_machine.BochaState.POSITIONING
            )
    else:
        time_below_threshold = 0.0

func handle_input(_event: InputEvent) -> void:
    pass

11.3 Paso 2: Modificar StateMachine.gd para usar @export

El StateMachine ya no crea los estados con .new(). Ahora los recibe como propiedades exportadas:

# StateMachine.gd — Versión Opción B
extends Node

signal state_changed(new_state: String)

enum BochaState {
    POSITIONING,
    AIMING,
    THROWING,
    ROLLING
}

var current_state: BochaState = BochaState.POSITIONING

# ⭐ Los estados se asignan desde el inspector, NO se crean con .new()
@export_group("States")
@export var positioning_state: PositioningState
@export var aiming_state: AimingState
@export var throwing_state: ThrowingState
@export var rolling_state: RollingState

var states: Dictionary = {}

func _ready() -> void:
    # Validar que todos los estados estén asignados
    if positioning_state == null:
        push_error("StateMachine: positioning_state no asignado en el inspector")
    if aiming_state == null:
        push_error("StateMachine: aiming_state no asignado en el inspector")
    if throwing_state == null:
        push_error("StateMachine: throwing_state no asignado en el inspector")
    if rolling_state == null:
        push_error("StateMachine: rolling_state no asignado en el inspector")

    # Mapear los estados al dictionary
    states[BochaState.POSITIONING] = positioning_state
    states[BochaState.AIMING] = aiming_state
    states[BochaState.THROWING] = throwing_state
    states[BochaState.ROLLING] = rolling_state

    # Inicializar cada estado
    for state in states.values():
        if state != null:
            state.init(get_parent())

    # Entrar al estado inicial
    if states[current_state] != null:
        states[current_state].enter()

func _process(delta: float) -> void:
    if states[current_state] != null:
        states[current_state].update(delta)

func _physics_process(delta: float) -> void:
    if states[current_state] != null:
        states[current_state].physics_update(delta)

func _input(event: InputEvent) -> void:
    if states[current_state] != null:
        states[current_state].handle_input(event)

func transition_to(new_state: BochaState) -> void:
    if new_state == current_state:
        return

    if states[current_state] != null:
        states[current_state].exit()

    current_state = new_state

    if states[current_state] != null:
        states[current_state].enter()
        state_changed.emit(BochaState.keys()[current_state])
